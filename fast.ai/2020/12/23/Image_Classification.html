def get_label(fname):
  if 'NORMAL' in fname.name:
    return 'NORMAL'
  else:
    return 'PNEUMONIA'

dblock = DataBlock(
         blocks=(ImageBlock, CategoryBlock),
         get_items=get_image_files,
         get_y=get_label,
         item_tfms=Resize(460),
         batch_tfms=aug_transforms(size=224, min_scale=0.75)
    )

train_normal_dblock = dblock.datasets(path/"train/NORMAL")
train_pneumonia_dblock = dblock.datasets(path/"train/PNEUMONIA")
